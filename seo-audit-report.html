⚠️ Critical SEO Issues Must Fix

These issues are directly preventing Google from ranking your pages. Fix immediately.

Query Parameter URLs — Not SEO Friendly
All your category/section pages use ?section= and ?group= parameters. Google can crawl these but they get no link equity and rank poorly vs. clean slug URLs. This is your biggest SEO problem.
✅ Fix: Implement URL Slug RedirectsRename pages to clean slugs OR add <link rel="canonical"> with slug-based URLs + update your .htaccess/Nginx to redirect. Delivered optimized redirect map below.
view.html — Same Canonical for ALL Sections
view.html has a static canonical pointing to /view.html regardless of which section is loaded. Google sees ALL sections as one duplicate page. This wastes 20+ ranking opportunities.
✅ Fix: Dynamic Canonical via JavaScriptSet canonical dynamically in JS based on ?section= value. Already delivered in optimized view.html file below.
Missing sitemap.xml / robots.txt
No sitemap detected in files. Without a sitemap, Google discovers pages slowly by crawling alone. Critical for a large jobs portal with 30+ section pages.
✅ Fix: Sitemap + Robots.txt DeliveredComplete sitemap.xml with all pages, lastmod dates, changefreq and priority delivered in this package.
Dynamic Title Tags on view.html
Title tag is static: "Top Sarkari Jobs | Latest Government Jobs" for every section page. Google uses title in rankings. "Railway Jobs 2026 | Top Sarkari Jobs" should be on the railway page.
✅ Fix: Section-specific titles via JSMap each section key → unique title + meta description + H1. Already included in optimized view.html.
Missing preconnect for External Resources

Proper schema markup enables rich results in Google — job cards, FAQs, breadcrumbs, sitelinks.

// JobPosting schema — inject dynamically in job.html when job data loads
function injectJobSchema(job) {
  const schema = {
    "@context": "https://schema.org",
    "@type": "JobPosting",
    "title": job.name || job.title,
    "description": job.description || `Apply for ${job.name} government job. Check eligibility, vacancies, and application details.`,
    "identifier": {
      "@type": "PropertyValue",
      "name": "Top Sarkari Jobs",
      "value": job.id || job.slug
    },
    "datePosted": job.postedDate || new Date().toISOString().split('T')[0],
    "validThrough": job.lastDate || "",
    "employmentType": "FULL_TIME",
    "hiringOrganization": {
      "@type": "Organization",
      "name": job.organization || "Government of India",
      "sameAs": "https://www.topsarkarijobs.com/"
    },
    "jobLocation": {
      "@type": "Place",
      "address": {
        "@type": "PostalAddress",
        "addressCountry": "IN",
        "addressRegion": job.state || "India"
      }
    },
    "baseSalary": job.salary ? {
      "@type": "MonetaryAmount",
      "currency": "INR",
      "value": { "@type": "QuantitativeValue", "value": job.salary, "unitText": "MONTH" }
    } : undefined,
    "jobBenefits": "Government job benefits, pension, allowances",
    "educationRequirements": job.qualification || ""
  };
  document.getElementById('jobSchema').textContent = JSON.stringify(schema);
}

// Dynamic BreadcrumbList for section pages
function injectBreadcrumb(sectionName, sectionUrl) {
  const schema = {
    "@context": "https://schema.org",
    "@type": "BreadcrumbList",
    "itemListElement": [
      {"@type":"ListItem","position":1,"name":"Home","item":"https://www.topsarkarijobs.com/"},
      {"@type":"ListItem","position":2,"name":sectionName,"item":"https://www.topsarkarijobs.com/" + sectionUrl}
    ]
  };
  const el = document.getElementById('breadcrumbSchema') || (() => {
    const s = document.createElement('script');
    s.type = 'application/ld+json';
    s.id = 'breadcrumbSchema';
    document.head.appendChild(s);
    return s;
  })();
  el.textContent = JSON.stringify(schema);
}
